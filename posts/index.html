    关于arch linux的安装教程#

    • 使用 setfont-132tar 设置终端大号字体，再进行网络连接，使用 ip link 检查网卡，通过iwctl工具进行无线网连接。
    • 最后使用 ping (website) 检查网络连接是否成功。
    • 随后使用fdisk工具对磁盘进行分区, 并格式化format分区，挂载mount分区。
    • 更换本地源origin可以提升下载速度降低失败率。
    • 现在可以使用pacstrap安装必要的固件。
    • 设置setting系统开机自动挂载规则并进入系统。
    • 设置setting时区并写入硬件 …

    fdisk

    关于fdisk工具的使用教程#

    fdisk -l  #列出可用硬盘和分区情况
fdisk device  #对指定设备进行分区
- g  #创建GPT分区
- n  #新建分区
- d  #删除分区
- t  #设置分区类型
- w  #保存分区

    format

    如何格式化分区#

    mkfs.ext4 /dev/root_partiion  #格式化根分区
mkswap /dev/swap_partition  #格式化交换分区
mkfs.fat -F32 /dev/efi_partition  #格式化efi分区

    grub 

    grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=GRUB  #把GRUB安装到EFI/磁盘
grub-mkconfig -o /boot/grub/grub.cfg  #生成启动菜单

    mount

    分区挂载教程#

    mount /dev/root /mnt  #挂载根分区
mount /dev/efi --mkdir /mnt/boot  #创建efi分区并挂载
sawpon /dev/swap_partition  #启动swap分区
